def self.create(props)
  obj = TestSend.new
  if props
    props.each do |key, value|
      if key == 'email_addresses'
        if value
          obj.email_addresses = []
          value.each do |email_address|
            obj.email_addresses << email_address
          end
        end
      else
        obj.send("#{key}=", value) if obj.respond_to? key
      end
    end
  end
  obj
end
