def self.create(props)
  obj = EventItem.new
  if props
    props.each do |key, value|
      if key == 'attributes'
        if value
          obj.attributes = []
          value.each do |attribute|
            obj.attributes << Components::EventItemAttribute.create(attribute)
          end
        end
      else
        obj.send("#{key}=", value) if obj.respond_to? key
      end
    end
  end
  obj
end
