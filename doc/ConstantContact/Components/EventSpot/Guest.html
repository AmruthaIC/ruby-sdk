def self.create(props)
  guest = Guest.new
  props.each do |key, value|
    key = key.to_s
    if key == 'guest_section'
      guest.send("#{key}=", Components::EventSpot::RegistrantSection.create(value))
    else
      guest.send("#{key}=", value) if guest.respond_to?("#{key}=")
    end
  end
  guest
end
